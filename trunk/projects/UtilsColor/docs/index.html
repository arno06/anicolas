The basics

Simple implementation for HSL <-> RGB color converter. 

var red = 255;
var green = 0;
var blue = 0;
var alpha = 1;

/**
 * Constructor from RGBA values
 **/
var color = new HSLColor(red, green, blue, alpha);

/**
 * Output : rgb(255, 0, 0)
 **/
console.log(color.getRGB());

/**
 * Output : rgba(255, 0, 0, 1)
 **/
console.log(color.getRGBA());

/**
 * Set Hue to 180 degree
 **/
color.setH(180);

/**
 * Set Saturation to 100 percent
 **/
color.setS(1);

/**
 * Set luminance to 50 percent
 **/
color.setL(0.5);
